-- Triggers to keep FTS index in sync
CREATE TRIGGER IF NOT EXISTS verses_ai AFTER INSERT ON verses BEGIN
INSERT INTO verse_search(content, osis_ref, translation_id)
VALUES (new.content, new.osis_ref, new.translation_id);
END;
CREATE TRIGGER IF NOT EXISTS verses_ad AFTER DELETE ON verses BEGIN
INSERT INTO verse_search(verse_search, rowid, content, osis_ref, translation_id)
VALUES ('delete', old.id, old.content, old.osis_ref, old.translation_id);
END;
